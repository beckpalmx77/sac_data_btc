SELECT SLMN_NAME,sum(CAST(TRD_G_KEYIN AS DECIMAL(10,2))) as  TRD_G_KEYIN
                                                      FROM ims_product_sale_sac 
                                                      WHERE DI_MONTH = '2'
                                                      AND DI_YEAR = '2024'
                                                      AND BRANCH = 'BTC'
                                                      GROUP BY  SLMN_NAME
                                                      ORDER BY SLMN_NAME
